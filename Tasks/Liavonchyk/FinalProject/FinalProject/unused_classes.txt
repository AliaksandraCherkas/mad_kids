class AddMaterial(forms.Form):
    type = forms.CharField(label='Type',empty_value = 'polymer clay')
    article = forms.CharField(label='Article (for FIMO please add 4 numbers of series e.g. 8020, dash and color)')
    manufacturer = forms.CharField(label='Manufacturer',empty_value = '')
    trade_mark = forms.CharField(label='Trade mark')
    series = forms.CharField(label='Series')
    color = forms.CharField(label='Color')
    effects = forms.CharField(widget=forms.Textarea(attrs={'class':"test-class"}), label='Effects')
    nominal_amount = forms.FloatField(label='Nominal amount')
    real_amount = forms.FloatField(label='Real amount')
    measure_units = forms.CharField(max_length=100, label='Measure units')
    unit_price_BYN = forms.FloatField(label='Unit price BYN')
#    unit_price_USD = forms.FloatField(label='Unit price USD')
    purchase_date = forms.DateTimeField(label='Purchase date')
    purchase_place = forms.CharField(label='Purchase place')
#    update_timestamp = forms.DateTimeField(label='Update timestamp')
    comment = forms.CharField(widget=forms.Textarea(attrs={'class':"test-class"}), label='Comment')
